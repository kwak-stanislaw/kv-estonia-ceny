repozytorium.model.trening

Trening modeli regresji ceny

Architektura:

. Wspólny preprocessing — imputacja braków

. Cel w skali log — TransformedTargetRegressor z log1p/expm1

. Dwa modele — regresja liniowa oraz Random Forest wychwytujący nieliniowości

:author: Stanisław Kwak :license: MIT

  6#. **Cel w skali log** — ``TransformedTargetRegressor`` z ``log1p``/``expm1``
  7#. **Dwa modele** — regresja liniowa oraz Random Forest wychwytujący nieliniowości

 66@dataclass
 67class PakietModeli:
 68    """Komplet rzeczy potrzebnych aplikacji do oceniania
 69
 70    Pakiet jest eksportowany do pliku ``joblib`` i wczytywany przez
 71    :class:`~repozytorium.model.predykcja.Estymator`
 72
 73    Attributes:
 74        model_liniowy: Wytrenowana regresja liniowa
 75        model_las: Wytrenowany Random Forest — nazwa pola zachowana dla pliku joblib
 76    """
 77
 78    model_liniowy: Pipeline
 79    model_las: Pipeline
 80    sigma_log: dict[str, float]
 81    metryki: dict[str, dict]
 82    cechy: tuple[str, ...]
 83    mediana_ceny: float
 84    wspolczynniki_liniowe: dict[str, float] = field(default_factory=dict)
 85    wersja: str = __version__
 86    autor: str = __author__
 87    licencja: str = __license__
 88    data_treningu: str = ""

 91class TrenerModelu:
 92
 93    def __init__(self, losowość: int = 42) -> None:
 94        """Ziarno generatora dla powtarzalności wyników"""
 95
 96        self.losowość = losowość
 97        self.budowniczy = BudowniczyCech()
 98
 99    def _preprocessor(self) -> ColumnTransformer:
100        """imputacja skalowanie one-hot"""
101
102        numeryczny = Pipeline(
103            steps=[
104                ("imputacja", SimpleImputer(strategy="median")),
105                ("skalowanie", StandardScaler()),
106            ]
107        )
108        kategoryczny = Pipeline(
109            steps=[
110                ("imputacja", SimpleImputer(strategy="most_frequent")),
111                ("kodowanie", OneHotEncoder(handle_unknown="ignore")),
112            ]
113        )
114        return ColumnTransformer(
115            transformers=[
116                ("num", numeryczny, list(schema.CECHY_NUMERYCZNE_PELNE)),
117                ("kat", kategoryczny, list(schema.CECHY_KATEGORYCZNE)),
118            ]
119        )
120
121    def _potok(self, regresor) -> Pipeline:
122        """Składa preprocessing z regresorem uczonym na logarytmie ceny"""
123
124        model_log = TransformedTargetRegressor(
125            regressor=regresor,
126            func=np.log1p,
127            inverse_func=np.expm1,
128        )
129        return Pipeline(
130            steps=[
131                ("preprocessing", self._preprocessor()),
132                ("model", model_log),
133            ]
134        )
135
136    def _ocen(
137        self,
138        nazwa: str,
139        potok: Pipeline,
140        X_test: pd.DataFrame,
141        y_test: pd.Series,
142        X: pd.DataFrame,
143        y: pd.Series,
144    ) -> tuple[WynikiModelu, float]:
145        """Liczy metryki wskali log"""
146
147        predykcje = potok.predict(X_test)
148        r2 = float(r2_score(y_test, predykcje))
149        mae = float(mean_absolute_error(y_test, predykcje))
150        rmse = float(root_mean_squared_error(y_test, predykcje))
151
152        cv = cross_val_score(potok, X, y, cv=5, scoring="r2")
153        wyniki = WynikiModelu(
154            nazwa=nazwa,
155            r2=r2,
156            mae=mae,
157            rmse=rmse,
158            r2_cv_srednia=float(cv.mean()),
159            r2_cv_odchylenie=float(cv.std()),
160        )
161
162        reszty_log = np.log1p(y_test.to_numpy()) - np.log1p(np.maximum(predykcje, 0))
163        sigma_log = float(np.std(reszty_log))
164        return wyniki, sigma_log
165
166    def _wspolczynniki_liniowe(self, potok: Pipeline) -> dict[str, float]:
167        """Wyciąga współczynniki regresji liniowej wraz z nazwami cech"""
168
169        try:
170            preprocessing = potok.named_steps["preprocessing"]
171            regresor = potok.named_steps["model"].regressor_
172            nazwy = preprocessing.get_feature_names_out()
173            wspolczynniki = regresor.coef_
174            return {
175                str(n): float(w) for n, w in zip(nazwy, wspolczynniki)
176            }
177        except Exception:  # pragma: no cover - diagnostyka pomocnicza
178            return {}
179
180    def trenuj(self, df: pd.DataFrame | None = None) -> PakietModeli:
181        """Trenuje oba modele i zwraca gotowy :class:`PakietModeli`
182
183        :param df: Dane wejściowe; gdy ``None`` — wczytywane przez
184            :class:`~repozytorium.data.loader.LadowarkaDanych`
185        :returns: Pakiet z modelami, metrykami i statystykami niepewności
186        """
187
188        if df is None:
189            df = LadowarkaDanych().wczytaj_do_modelu()
190
191        X, y = self.budowniczy.przygotuj_X_y(df)
192        X_train, X_test, y_train, y_test = train_test_split(
193            X, y, test_size=0.2, random_state=self.losowość
194        )
195
196        potok_liniowy = self._potok(LinearRegression())
197        potok_las = self._potok(
198            RandomForestRegressor(
199                n_estimators=300,
200                max_depth=None,
201                min_samples_leaf=2,
202                n_jobs=-1,
203                random_state=self.losowość,
204            )
205        )
206
207        potok_liniowy.fit(X_train, y_train)
208        potok_las.fit(X_train, y_train)
209
210        wyniki_lin, sigma_lin = self._ocen(
211            "Regresja liniowa", potok_liniowy, X_test, y_test, X, y
212        )
213        wyniki_las, sigma_las = self._ocen(
214            "Random Forest", potok_las, X_test, y_test, X, y
215        )
216
217        return PakietModeli(
218            model_liniowy=potok_liniowy,
219            model_las=potok_las,
220            sigma_log={"liniowy": sigma_lin, "las": sigma_las},
221            metryki={
222                "liniowy": wyniki_lin.jako_slownik(),
223                "las": wyniki_las.jako_slownik(),
224            },
225            cechy=schema.WSZYSTKIE_CECHY,
226            mediana_ceny=float(y.median()),
227            wspolczynniki_liniowe=self._wspolczynniki_liniowe(potok_liniowy),
228            data_treningu=datetime.now(timezone.utc).isoformat(timespec="seconds"),
229        )
230
231    def trenuj_i_zapisz(
232        self, sciezka: str | Path | None = None
233    ) -> tuple[PakietModeli, Path]:
234        """Trenuje modele i zapisuje pakiet na dysk
235
236        :param sciezka: Docelowy plik ``.joblib``; gdy ``None`` — domyślny
237            z :func:`~repozytorium.paths.default_model_bundle_path`
238        :returns: Pakiet modeli i ścieżka zapisu
239        """
240
241        pakiet = self.trenuj()
242        cel = Path(sciezka) if sciezka else default_model_bundle_path()
243        cel.parent.mkdir(parents=True, exist_ok=True)
244        joblib.dump(pakiet, cel)
245        return pakiet, cel
